\newcommand*\circled[1]{%commande independante des autres : encercler #1
	\tikz[baseline=(char.base)]{%
		\node[shape=circle,fill=white,draw,inner sep=1pt,opacity=1] (char) {#1};%rotate=90
	}
}

\newcommand{\pts}[2]{%commande independante des autres : placer#2 avec distance #1 avant la position du curseur 
	\tikz[overlay,anchor=base, baseline=(4.base),xshift=-#1cm] {\node[blue](4){\textbf{#2}};}
}
